```python
import os
import requests
import time
import logging
import re
from typing import Optional, Dict, Any

API_KEY = os.getenv("CREWAI_API_KEY")
API_URL = "https://api.crewai.com/v1/tools/sentiment-analysis"

# Setup logging
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

def is_valid_text(input_text: str) -> bool:
    # Basic validation: only alphanumeric, spaces, and punctuation allowed
    pattern = r'^[\w\s.,!?\'"-]+$'
    return bool(re.fullmatch(pattern, input_text))

def send_request_with_retries(text: str, retries: int = 3, delay: int = 2) -> Optional[Dict[str, Any]]:
    if not API_KEY:
        logging.error("CREWAI_API_KEY environment variable not set.")
        return None

    if not is_valid_text(text):
        logging.warning("Input validation failed: Contains invalid characters.")
        return None

    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json"
    }
    payload = {"text": text}

    for attempt in range(1, retries + 1):
        try:
            logging.info(f"Attempt {attempt} sending request...")
            response = requests.post(API_URL, json=payload, headers=headers, timeout=10)
            response.raise_for_status()

            result = response.json()
            logging.info(f"Received result: {result}")
            return result

        except requests.exceptions.RequestException as e:
            logging.error(f"Request failed: {e}")
            if attempt < retries:
                logging.info(f"Retrying after {delay} seconds...")
                time.sleep(delay)
            else:
                logging.error("Max retries reached, aborting.")
                return None

if __name__ == "__main__":

    if result:
```

```python
import os
import requests
import time
import logging
import re
from typing import Optional, Dict, Any

# Configure logging
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

API_KEY = os.getenv("CREWAI_API_KEY")
API_URL = "https://api.crewai.com/v1/tools/sentiment-analysis"

def is_valid_text(text: str) -> bool:
    """
    Validate the input text to allow only alphanumeric characters, spaces,
    and specific punctuation: . , ! ? ' " -
    """
    pattern = r'^[\w\s.,!?\'"-]+$'
    return bool(re.fullmatch(pattern, text))

def send_sentiment_request(text: str, retries: int = 3, delay: int = 2) -> Optional[Dict[str, Any]]:
    """
    Send the text to the CrewAI sentiment analysis API endpoint
    with retry logic for robustness.
    """
    if not API_KEY:
        logging.error("CREWAI_API_KEY environment variable not set.")
        return None

    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json"
    }
    payload = {"text": text}

    for attempt in range(1, retries + 1):
        try:
            logging.info(f"Attempt {attempt}: Sending request for sentiment analysis.")
            response = requests.post(API_URL, json=payload, headers=headers, timeout=10)
            response.raise_for_status()
            logging.info(f"Response status code: {response.status_code}")
            return response.json()
        except requests.exceptions.RequestException as e:
            logging.error(f"Request failed: {e}")
            if attempt < retries:
                logging.info(f"Retrying after {delay} seconds...")
                time.sleep(delay)
            else:
                logging.error("Max retries reached. Aborting request.")
                return None

def main() -> None:
    user_input = input("Enter text to analyze sentiment: ").strip()

    if not is_valid_text(user_input):
        print("Input contains invalid characters. Please enter alphanumeric characters and basic punctuation only.")
        return

    result = send_sentiment_request(user_input)
    if result:
        sentiment = result.get("sentiment", "unknown")
        score = result.get("score", "N/A")
        print(f"Sentiment: {sentiment} (Score: {score})")
    else:
        print("Failed to retrieve sentiment analysis after multiple attempts.")

if __name__ == "__main__":
    main()
```
